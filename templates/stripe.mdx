## Configure Webhooks

<Steps>
  <Step title="Find Webhooks">
    Search for `webhook` in the Stripe dashboard
    <img src="/images/stripe/stripe-find-webhook-menu.png" style={{ borderRadius: '0.5rem' }} />
  </Step>
  
  <Step title="Create Endpoint">
    <img src="/images/stripe/stripe-create-webhook.png" style={{ borderRadius: '0.5rem' }} />
  </Step>
  
  <Step title="Set Endpoint URL">
    <Info>Use the format: `your-app.created.app/api/stripe-webhook`</Info>
    <img src="/images/stripe/stripe-set-webhook-url.png" style={{ borderRadius: '0.5rem' }} />
  </Step>
  
  <Step title="Get Project ID">
    <img src="/images/stripe/stripe-get-project-id.png" style={{ borderRadius: '0.5rem' }} />
  </Step>
  
  <Step title="Configure Events">
    <img src="/images/stripe/stripe-webhook-events-menu.png" style={{ borderRadius: '0.5rem' }} />
  </Step>
  
  <Step title="Select Events">
    Subscribe to these events:
    - `customer.subscription.created`
    - `customer.subscription.updated`
    
    <img src="/images/stripe/stripe-select-subscription-events.png" style={{ borderRadius: '0.5rem' }} />
  </Step>
  
  <Step title="Verify Configuration">
    Your webhook configuration should look like this:
    <img src="/images/stripe/stripe-webhook-final-form.png" style={{ borderRadius: '0.5rem' }} />
  </Step>
  
  <Step title="Get Webhook Secret">
    <img src="/images/stripe/stripe-reveal-webhook-secret.png" style={{ borderRadius: '0.5rem' }} />
    <img src="/images/stripe/stripe-copy-webhook-secret.png" style={{ borderRadius: '0.5rem' }} />
  </Step>
  
  <Step title="Configure Create">
    Add the webhook secret to your Create configuration
    <img src="/images/stripe/stripe-set-webhook-secret-create.png" style={{ borderRadius: '0.5rem' }} />
  </Step>
</Steps>
